opioid_by_race <- read.csv("~/SURE-2022/ODbyRace.csv")
opioid_by_age <- read.csv("~/SURE-2022/ODbyAge.csv")
opioid_by_sex <- read.csv("~/SURE-2022/ODbyGender.csv")
opioid_by_type <- read.csv("~/SURE-2022/ODbyType.csv")
o1 <- dplyr::left_join(x = opioid_by_age, y = opioid_by_race, by = 'State')
o2 <- dplyr::left_join(x = o1, y = opioid_by_sex, by = 'State')
opioid <- dplyr::left_join(x = o2, y = opioid_by_type, by = 'State')
opioid <- opioid %>%
  filter(!State %in% "United States")

opioid <- opioid %>%
  mutate(std_natural = as.numeric(scale(Natural.and.Semisynthetic.Opioids..e.g..oxycodone..hydrocodone., center = TRUE, scale = TRUE)),
         std_synthetic = as.numeric(scale(Synthetic.Opioids..other.than.Methadone..e.g..fentanyl..tramadol., center = TRUE, scale = TRUE)))
